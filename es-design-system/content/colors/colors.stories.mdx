import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs';
import { bsVariables } from '@/utils/metadata';
import { kebabCase } from '@/utils/case';

<Meta
    title="Documentation/Colors"
    parameters={{
        viewMode: 'docs',
        previewTabs: {
            canvas: { hidden: true },
        },
    }}
/>

# Variants

<ColorPalette>
    {Object.keys(bsVariables.themeColors).map((key) => {
        if (typeof bsVariables.themeColors[key] === 'string') {
            return (
                <ColorItem
                    title={kebabCase(key)}
                    colors={[bsVariables.themeColors[key]]}
                />
            );
        }
        return false;
    })}
</ColorPalette>
